## Тесты для `FileSystemExecutor` (path traversal)

Добавить в `tests/client/test_file_system_executor.py`:

```python
import asyncio
import pytest
from pathlib import Path
from codelab.client.infrastructure.services.file_system_executor import FileSystemExecutor


@pytest.fixture
def sandbox(tmp_path: Path) -> Path:
    (tmp_path / "safe.txt").write_text("safe content")
    (tmp_path / "subdir").mkdir()
    (tmp_path / "subdir" / "nested.txt").write_text("nested")
    return tmp_path


class TestPathTraversalProtection:

    def test_dotdot_traversal_blocked(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        with pytest.raises(ValueError, match="Path traversal"):
            executor._validate_path("../../../etc/passwd")

    def test_absolute_path_outside_blocked(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        with pytest.raises(ValueError, match="Path traversal"):
            executor._validate_path("/etc/passwd")

    def test_similar_name_directory_blocked(self, sandbox):
        """Директория /sandbox_evil не является дочерней к /sandbox."""
        evil_dir = sandbox.parent / (sandbox.name + "_evil")
        evil_dir.mkdir(exist_ok=True)
        (evil_dir / "secret.txt").write_text("secret")

        executor = FileSystemExecutor(base_path=sandbox)
        with pytest.raises(ValueError, match="Path traversal"):
            executor._validate_path(str(evil_dir / "secret.txt"))
        evil_dir.rmdir() if not list(evil_dir.iterdir()) else None

    def test_nested_dotdot_traversal_blocked(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        with pytest.raises(ValueError, match="Path traversal"):
            executor._validate_path("subdir/../../etc/passwd")

    def test_valid_relative_path_allowed(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        result = executor._validate_path("safe.txt")
        assert result == (sandbox / "safe.txt").resolve()

    def test_valid_nested_path_allowed(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        result = executor._validate_path("subdir/nested.txt")
        assert result == (sandbox / "subdir" / "nested.txt").resolve()

    @pytest.mark.asyncio
    async def test_read_outside_sandbox_blocked(self, sandbox):
        executor = FileSystemExecutor(base_path=sandbox)
        with pytest.raises(ValueError, match="Path traversal"):
            await executor.read_text_file("../../../etc/hostname")

    @pytest.mark.asyncio
    async def test_write_outside_sandbox_blocked(self, sandbox, tmp_path):
        executor = FileSystemExecutor(base_path=sandbox)
        target = tmp_path / "outside.txt"
        with pytest.raises(ValueError, match="Path traversal"):
            await executor.write_text_file(str(target), "malicious content")
        # Файл не должен быть создан
        assert not target.exists()
```

## Тесты для `TerminalExecutor` (shell injection)

Добавить в `tests/client/test_terminal_executor.py`:

```python
import pytest
from codelab.client.infrastructure.services.terminal_executor import TerminalExecutor


class TestShellInjectionProtection:

    def test_semicolon_not_interpreted(self):
        """Точка с запятой не должна создавать две команды."""
        executor = TerminalExecutor()
        # С shell=False "echo safe; echo injected" — это команда "echo"
        # с аргументом "safe; echo injected"
        result = executor.execute("echo safe; echo injected")
        # Обе строки должны быть в одном аргументе, не как отдельные команды
        assert result["success"] is True
        assert "safe; echo injected" in result["output"]

    def test_pipe_not_interpreted(self):
        """Пайп не должен перенаправлять вывод."""
        executor = TerminalExecutor()
        result = executor.execute("echo hello | cat")
        # "hello | cat" — это аргументы echo, не перенаправление
        assert "hello | cat" in result["output"]

    def test_subshell_not_executed(self):
        """Подоболочка $(...) не должна выполняться."""
        executor = TerminalExecutor()
        result = executor.execute("echo $(id)")
        # Должен вывести литеральную строку "$(id)", а не результат выполнения id
        assert "$(id)" in result["output"]
        assert "uid=" not in result["output"]

    def test_background_process_not_spawned(self):
        """Оператор & не должен создавать фоновый процесс."""
        executor = TerminalExecutor()
        result = executor.execute("echo test & echo background")
        # "test & echo background" — аргументы echo
        assert "test & echo background" in result["output"]

    def test_command_not_found_returns_127(self):
        executor = TerminalExecutor()
        result = executor.execute("nonexistent_cmd_xyz_abc_12345")
        assert result["exit_code"] == 127
        assert result["success"] is False
```

## Тесты для аутентификации (brute force protection)

После реализации rate limiting (задача 3.8) добавить:

```python
# tests/server/test_auth_rate_limit.py
import pytest
from codelab.server.protocol.core import ACPProtocol
from codelab.server.messages import ACPMessage


@pytest.mark.asyncio
async def test_multiple_failed_auth_attempts_are_rate_limited():
    """После N неудачных попыток — cooldown."""
    protocol = ACPProtocol(require_auth=True, auth_api_key="correct-key")

    # Инициализация
    await protocol.handle(ACPMessage.request("initialize", {
        "protocolVersion": 1,
        "clientCapabilities": {},
    }))

    # 5 неудачных попыток
    for _ in range(5):
        outcome = await protocol.handle(ACPMessage.request("authenticate", {
            "methodId": "local",
            "apiKey": "wrong-key",
        }))
        assert outcome.response.error is not None

    # 6-я попытка должна быть заблокирована (rate limit)
    outcome = await protocol.handle(ACPMessage.request("authenticate", {
        "methodId": "local",
        "apiKey": "correct-key",  # даже правильный ключ
    }))
    assert outcome.response.error is not None
    assert outcome.response.error.code == -32429  # Too Many Requests
```
