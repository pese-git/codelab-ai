# 3.7 — Rate Limiting для метода `authenticate`

**Приоритет:** 🟡 Средний  
**Оценка:** 4 часа  
**Файлы:** `src/codelab/server/protocol/handlers/auth.py`, `src/codelab/server/protocol/core.py`

---

## Проблема

Метод `authenticate` не защищён от brute force. Атакующий может отправлять тысячи запросов с разными ключами, пока не угадает правильный. Ограничений на число попыток нет.

---

## Решение: экспоненциальный backoff с lockout

### Шаг 1 — Создать `AuthRateLimiter`

Создать `src/codelab/server/protocol/handlers/auth_rate_limiter.py`:

```python
"""Rate limiter для защиты метода authenticate от brute force."""

from __future__ import annotations

import asyncio
import time
from dataclasses import dataclass, field

import structlog

logger = structlog.get_logger()

# Конфигурация по умолчанию
MAX_ATTEMPTS = 5             # После 5 неудач — lockout
LOCKOUT_SECONDS = 30.0       # Базовое время блокировки
MAX_LOCKOUT_SECONDS = 3600.0 # Максимальная блокировка (1 час)
WINDOW_SECONDS = 300.0       # Окно для подсчёта попыток (5 минут)


@dataclass
class AuthAttemptState:
    failed_count: int = 0
    locked_until: float = 0.0
    lockout_multiplier: int = 1
    first_attempt_at: float = field(default_factory=time.monotonic)


class AuthRateLimiter:
    """Трекер неудачных попыток аутентификации с экспоненциальным backoff.

    Привязан к протокольному инстансу (per-connection).
    """

    def __init__(
        self,
        max_attempts: int = MAX_ATTEMPTS,
        base_lockout: float = LOCKOUT_SECONDS,
        max_lockout: float = MAX_LOCKOUT_SECONDS,
        window: float = WINDOW_SECONDS,
    ) -> None:
        self._max_attempts = max_attempts
        self._base_lockout = base_lockout
        self._max_lockout = max_lockout
        self._window = window
        self._state = AuthAttemptState()

    def is_locked(self) -> bool:
        """True если сейчас действует блокировка."""
        return time.monotonic() < self._state.locked_until

    def remaining_lockout(self) -> float:
        """Оставшееся время блокировки в секундах."""
        return max(0.0, self._state.locked_until - time.monotonic())

    def record_failure(self) -> None:
        """Зафиксировать неудачную попытку. Установить lockout если превышен лимит."""
        now = time.monotonic()
        state = self._state

        # Сбросить счётчик если окно устарело
        if now - state.first_attempt_at > self._window:
            state.failed_count = 0
            state.first_attempt_at = now
            state.lockout_multiplier = 1

        state.failed_count += 1
        logger.warning(
            "auth_failed",
            attempt=state.failed_count,
            max_attempts=self._max_attempts,
        )

        if state.failed_count >= self._max_attempts:
            # Экспоненциальный backoff: 30s → 60s → 120s → ... → 3600s
            lockout_duration = min(
                self._base_lockout * state.lockout_multiplier,
                self._max_lockout,
            )
            state.locked_until = now + lockout_duration
            state.lockout_multiplier = min(state.lockout_multiplier * 2, 64)
            state.failed_count = 0  # Сброс счётчика после lockout

            logger.warning(
                "auth_locked_out",
                duration_seconds=lockout_duration,
                multiplier=state.lockout_multiplier,
            )

    def record_success(self) -> None:
        """Сбросить счётчик неудач после успешной аутентификации."""
        self._state = AuthAttemptState()
        logger.debug("auth_success_rate_limit_reset")
```

### Шаг 2 — Добавить в `ACPProtocol`

```python
# core.py
from .handlers.auth_rate_limiter import AuthRateLimiter

class ACPProtocol:
    def __init__(self, ...) -> None:
        ...
        self._auth_rate_limiter = AuthRateLimiter()
```

### Шаг 3 — Использовать в методе `authenticate`

```python
# В ACPProtocol.handle() или _handle_authenticate():

if method == "authenticate":
    # Проверить rate limit ДО обработки
    if self._auth_rate_limiter.is_locked():
        remaining = self._auth_rate_limiter.remaining_lockout()
        return ProtocolOutcome(
            response=ACPMessage.error_response(
                message.id,
                code=-32429,  # Too Many Requests (нестандартный, но понятный)
                message="Too many failed attempts. Try again later.",
                data={"retry_after_seconds": round(remaining)},
            )
        )

    response, authenticated = auth.authenticate(
        message.id, params, self._require_auth, self._auth_api_key, self._auth_methods,
    )

    if authenticated:
        self._authenticated = True
        self._auth_rate_limiter.record_success()
    else:
        # Зафиксировать неудачу только если ключ был неверным
        # (не для ошибок валидации параметров)
        if response.error and response.error.code == -32011:  # auth_failed
            self._auth_rate_limiter.record_failure()

    return ProtocolOutcome(response=response)
```

## Тесты

```python
import pytest
import time
from unittest.mock import patch
from codelab.server.protocol.handlers.auth_rate_limiter import AuthRateLimiter


def test_no_lockout_initially():
    limiter = AuthRateLimiter(max_attempts=3, base_lockout=10.0)
    assert not limiter.is_locked()


def test_lockout_after_max_attempts():
    limiter = AuthRateLimiter(max_attempts=3, base_lockout=10.0)
    limiter.record_failure()
    limiter.record_failure()
    assert not limiter.is_locked()
    limiter.record_failure()
    assert limiter.is_locked()


def test_lockout_expires():
    limiter = AuthRateLimiter(max_attempts=1, base_lockout=0.1)  # 100ms lockout
    limiter.record_failure()
    assert limiter.is_locked()
    time.sleep(0.15)
    assert not limiter.is_locked()


def test_success_resets_counter():
    limiter = AuthRateLimiter(max_attempts=3, base_lockout=10.0)
    limiter.record_failure()
    limiter.record_failure()
    limiter.record_success()
    # После успеха — счётчик сброшен
    limiter.record_failure()
    limiter.record_failure()
    assert not limiter.is_locked()  # нужно ещё 1 попытку до лимита


def test_exponential_backoff():
    limiter = AuthRateLimiter(max_attempts=2, base_lockout=10.0, max_lockout=1000.0)

    # Первый lockout
    limiter.record_failure()
    limiter.record_failure()
    remaining1 = limiter.remaining_lockout()
    assert 9.0 < remaining1 <= 10.0

    # Сбросить lockout через прямой сброс (для теста)
    limiter._state.locked_until = 0

    # Второй lockout — в 2 раза длиннее
    limiter.record_failure()
    limiter.record_failure()
    remaining2 = limiter.remaining_lockout()
    assert remaining2 > remaining1  # экспоненциальный рост
```
