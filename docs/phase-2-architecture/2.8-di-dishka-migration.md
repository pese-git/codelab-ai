**Приоритет:** 🟡 Средний  
**Оценка:** 2 дня  
**Файлы:** `src/codelab/client/infrastructure/di_container.py`, `di_bootstrapper.py`, `presentation/view_model_factory.py`

## Проблема

Текущий DI-контейнер (~440 строк суммарно) содержит несколько принципиальных изъянов:

| Проблема | Проявление |
|---|---|
| Не делает DI — только хранит объекты | Все сервисы создаются вручную в bootstrapper |
| SCOPED = SINGLETON (TODO) | Ложный контракт, нет per-request изоляции |
| `dispose()` синхронный | Async-ресурсы (WebSocket) не закрываются корректно |
| Циклическая зависимость через мутацию приватного поля | `coordinator._permission_handler = permission_handler` |
| `# type: ignore[call-top-callable]` | Несовместимость типов в `Registration.create()` |

**Решение:** заменить на [`dishka`](https://github.com/reagento/dishka) — async-first DI для Python с правильными scopes и lifecycle через `AsyncContextManager`.

## Шаг 2 — Разорвать циклическую зависимость

Перед миграцией — убрать цикл `SessionCoordinator ↔ PermissionHandler` через `EventBus`:

```python
# БЫЛО: coordinator._permission_handler = permission_handler (после создания)

# СТАЛО: оба используют EventBus для общения:
# SessionCoordinator публикует события → PermissionHandler слушает
# PermissionHandler публикует решения → SessionCoordinator слушает
class SessionCoordinator:
    def __init__(self, transport: ACPTransportService, session_repo, event_bus: EventBus):
        self._event_bus = event_bus
        # Больше нет прямой ссылки на PermissionHandler

class PermissionHandler:
    def __init__(self, transport: ACPTransportService, event_bus: EventBus):
        self._event_bus = event_bus
        # Больше нет прямой ссылки на SessionCoordinator
```

## Шаг 3 — Создать `AppProvider`

Создать `src/codelab/client/infrastructure/app_provider.py`:

```python
"""dishka Provider — декларативная конфигурация зависимостей клиентского приложения."""

from __future__ import annotations

import os
from collections.abc import AsyncIterator
from pathlib import Path
from typing import Any

import structlog
from dishka import Provider, Scope, provide

from codelab.client.application.permission_handler import PermissionHandler
from codelab.client.application.session_coordinator import SessionCoordinator
from codelab.client.domain.repositories import SessionRepository
from codelab.client.infrastructure.events.bus import EventBus
from codelab.client.infrastructure.handlers.file_system_handler import FileSystemHandler
from codelab.client.infrastructure.handlers.terminal_handler import TerminalHandler
from codelab.client.infrastructure.repositories import InMemorySessionRepository
from codelab.client.infrastructure.services.acp_transport_service import ACPTransportService
from codelab.client.infrastructure.services.file_system_executor import FileSystemExecutor
from codelab.client.infrastructure.services.terminal_executor import TerminalExecutor
from codelab.client.presentation.chat_view_model import ChatViewModel
from codelab.client.presentation.file_viewer_view_model import FileViewerViewModel
from codelab.client.presentation.filesystem_view_model import FileSystemViewModel
from codelab.client.presentation.permission_view_model import PermissionViewModel
from codelab.client.presentation.plan_view_model import PlanViewModel
from codelab.client.presentation.session_view_model import SessionViewModel
from codelab.client.presentation.terminal_log_view_model import TerminalLogViewModel
from codelab.client.presentation.terminal_view_model import TerminalViewModel
from codelab.client.presentation.ui_view_model import UIViewModel

logger = structlog.get_logger("app_provider")


class AppProvider(Provider):
    """Декларативный провайдер всех зависимостей приложения.

    Dishka автоматически разрешает зависимости по аннотациям типов.
    Async lifecycle (AsyncIterator) гарантирует корректное закрытие ресурсов.
    """

    scope = Scope.APP  # Все сервисы — singleton на время жизни приложения

    def __init__(self, host: str, port: int, cwd: str, history_dir: str | None = None):
        super().__init__()
        self._host = host
        self._port = port
        self._cwd = cwd
        self._history_dir = history_dir

    # --- Инфраструктурные сервисы ---

    @provide
    def event_bus(self) -> EventBus:
        return EventBus()

    @provide
    async def transport(self, event_bus: EventBus) -> AsyncIterator[ACPTransportService]:
        """Транспортный сервис с гарантированным async закрытием."""
        service = ACPTransportService(
            host=self._host,
            port=self._port,
            event_bus=event_bus,
        )
        yield service
        # Этот код выполнится при завершении приложения:
        await service.close()
        logger.debug("transport_closed")

    @provide
    def session_repository(self) -> InMemorySessionRepository:
        return InMemorySessionRepository()

    @provide
    def fs_executor(self) -> FileSystemExecutor:
        return FileSystemExecutor(base_path=Path(self._cwd))

    @provide
    async def terminal_executor(self) -> AsyncIterator[TerminalExecutor]:
        executor = TerminalExecutor()
        yield executor
        await executor.cleanup_all()
        logger.debug("terminal_executor_cleaned_up")

    # --- Обработчики ---

    @provide
    def fs_handler(
        self,
        executor: FileSystemExecutor,
        permission_handler: PermissionHandler,
    ) -> FileSystemHandler:
        return FileSystemHandler(executor=executor, permission_handler=permission_handler)

    @provide
    def terminal_handler(self, executor: TerminalExecutor) -> TerminalHandler:
        return TerminalHandler(executor)

    # --- Application слой ---

    @provide
    def session_coordinator(
        self,
        transport: ACPTransportService,
        session_repo: InMemorySessionRepository,
        event_bus: EventBus,
    ) -> SessionCoordinator:
        """Цикл разорван: SessionCoordinator общается через EventBus."""
        return SessionCoordinator(
            transport=transport,
            session_repo=session_repo,
            event_bus=event_bus,
        )

    @provide
    def permission_handler(
        self,
        transport: ACPTransportService,
        event_bus: EventBus,
    ) -> PermissionHandler:
        """Цикл разорван: PermissionHandler общается через EventBus."""
        return PermissionHandler(transport=transport, event_bus=event_bus)

    # --- ViewModels ---

    @provide
    def ui_view_model(self, event_bus: EventBus) -> UIViewModel:
        return UIViewModel(event_bus=event_bus, logger=logger)

    @provide
    def session_view_model(
        self, coordinator: SessionCoordinator, event_bus: EventBus
    ) -> SessionViewModel:
        return SessionViewModel(coordinator=coordinator, event_bus=event_bus, logger=logger)

    @provide
    def plan_view_model(self, event_bus: EventBus) -> PlanViewModel:
        return PlanViewModel(event_bus=event_bus, logger=logger)

    @provide
    def chat_view_model(
        self,
        coordinator: SessionCoordinator,
        event_bus: EventBus,
        fs_executor: FileSystemExecutor,
        terminal_executor: TerminalExecutor,
        plan_vm: PlanViewModel,
    ) -> ChatViewModel:
        return ChatViewModel(
            coordinator=coordinator,
            event_bus=event_bus,
            logger=logger,
            history_dir=self._history_dir,
            fs_executor=fs_executor,
            terminal_executor=terminal_executor,
            plan_vm=plan_vm,
        )

    @provide
    def terminal_view_model(self, event_bus: EventBus) -> TerminalViewModel:
        return TerminalViewModel(event_bus=event_bus, logger=logger)

    @provide
    def filesystem_view_model(self, event_bus: EventBus) -> FileSystemViewModel:
        return FileSystemViewModel(event_bus=event_bus, logger=logger)

    @provide
    def file_viewer_view_model(self, event_bus: EventBus) -> FileViewerViewModel:
        return FileViewerViewModel(event_bus=event_bus, logger=logger)

    @provide
    def permission_view_model(self, event_bus: EventBus) -> PermissionViewModel:
        return PermissionViewModel(event_bus=event_bus, logger=logger)

    @provide
    def terminal_log_view_model(self, event_bus: EventBus) -> TerminalLogViewModel:
        return TerminalLogViewModel(event_bus=event_bus, logger=logger)
```

## Шаг 4 — Обновить точку входа приложения

```python
# src/codelab/client/tui/__main__.py или app.py

from dishka import make_async_container
from codelab.client.infrastructure.app_provider import AppProvider
from codelab.client.presentation.ui_view_model import UIViewModel
from codelab.client.presentation.chat_view_model import ChatViewModel

async def run_app(host: str, port: int, cwd: str) -> None:
    provider = AppProvider(host=host, port=port, cwd=cwd)
    container = make_async_container(provider)

    async with container() as ctx:
        # Получить нужные сервисы из контейнера
        ui_vm = await ctx.get(UIViewModel)
        chat_vm = await ctx.get(ChatViewModel)
        # ... при выходе из `async with` все AsyncIterator ресурсы закроются автоматически

        app = ACPClientApp(ui_vm=ui_vm, chat_vm=chat_vm, ...)
        await app.run_async()
```

## Шаг 6 — Обновить тесты

В тестах можно переопределять отдельные зависимости через `TestProvider`:

from dishka import Provider, Scope, provide, make_async_container
from unittest.mock import AsyncMock, MagicMock

class TestProvider(Provider):
    scope = Scope.APP

    @provide
    def transport(self) -> ACPTransportService:
        mock = MagicMock(spec=ACPTransportService)
        mock.send = AsyncMock()
        return mock

    @provide
    def event_bus(self) -> EventBus:
        return EventBus()  # Реальный, не mock


@pytest.fixture
async def container():
    app_provider = AppProvider(host="localhost", port=8765, cwd="/tmp")
    test_provider = TestProvider()
    c = make_async_container(app_provider, test_provider)  # test перекрывает app
    async with c() as ctx:
        yield ctx

## Итог миграции

| До | После |
|---|---|
| ~440 строк кода | ~100 строк (`AppProvider`) |
| SCOPED = SINGLETON (не реализовано) | Правильные scopes APP / REQUEST / SESSION |
| Синхронный `dispose()`, пропускает async-ресурсы | Гарантированный async cleanup через `AsyncIterator` |
| Циклическая зависимость через мутацию поля | Цикл разорван через EventBus |
| `# type: ignore[call-top-callable]` | Полная типизация, нет подавлений |
| Объекты создаются вручную в bootstrapper | Auto-wiring по аннотациям типов |
