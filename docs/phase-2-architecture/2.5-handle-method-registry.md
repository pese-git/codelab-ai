# 2.5 — Замена цепочки `if method ==` на реестр обработчиков в `ACPProtocol.handle()`

**Приоритет:** 🟡 Средний  
**Оценка:** 1 день  
**Файл:** `src/codelab/server/protocol/core.py`

---

## Проблема

Метод `ACPProtocol.handle()` содержит 400+ строк с цепочкой `if method == "..."` на 15 ветвей. Каждая ветвь содержит не только делегирование, но и побочную логику (обновление capabilities, инициализация MCP и т.д.). Логика инициализации MCP-серверов дублируется в `session/new` и `session/load`.

---

## Решение

Ввести реестр обработчиков. Каждый обработчик — отдельный метод или класс.

### Шаг 1 — Определить тип обработчика

```python
# core.py
from typing import Callable, Awaitable
from .messages import ACPMessage
from .protocol.state import ProtocolOutcome

# Тип: async-функция, принимающая сообщение и возвращающая outcome
MethodHandler = Callable[[ACPMessage], Awaitable[ProtocolOutcome]]
```

### Шаг 2 — Зарегистрировать обработчики в `__init__`

```python
class ACPProtocol:
    def __init__(self, ...) -> None:
        ...
        # Реестр обработчиков методов
        self._handlers: dict[str, MethodHandler] = {
            "initialize":                           self._handle_initialize,
            "authenticate":                         self._handle_authenticate,
            "session/new":                          self._handle_session_new,
            "session/load":                         self._handle_session_load,
            "session/list":                         self._handle_session_list,
            "session/prompt":                       self._handle_session_prompt,
            "session/cancel":                       self._handle_session_cancel,
            "session/request_permission_response":  self._handle_permission_response,
            "session/set_config_option":            self._handle_set_config_option,
            "session/set_mode":                     self._handle_set_mode,
            "ping":                                 self._handle_ping,
            "echo":                                 self._handle_echo,
            "shutdown":                             self._handle_shutdown,
        }
```

### Шаг 3 — Упростить `handle()`

```python
async def handle(self, message: ACPMessage) -> ProtocolOutcome:
    """Маршрутизировать входящее сообщение по методу."""
    # Ответы от клиента (method=None) — отдельный путь
    if message.method is None:
        return self.handle_client_response(message)

    method = message.method
    handler = self._handlers.get(method)

    if handler is None:
        # Уведомления — игнорируем без ошибки
        if message.is_notification:
            return ProtocolOutcome()
        return ProtocolOutcome(
            response=ACPMessage.error_response(
                message.id, code=-32601, message=f"Method not found: {method}"
            )
        )

    return await handler(message)
```

### Шаг 4 — Вынести логику в отдельные методы

Пример для `session/new` и `session/load` — вынести общую логику MCP в приватный метод:

```python
async def _handle_session_new(self, message: ACPMessage) -> ProtocolOutcome:
    params = message.params or {}
    response_msg = session.session_new(
        message.id, params, self._require_auth, self._authenticated,
        self._config_specs, self._auth_methods, self._runtime_capabilities,
    )
    if response_msg.result is not None:
        session_id = response_msg.result.get("sessionId")
        if isinstance(session_id, str):
            session_state = SessionFactory.create_session(
                cwd=params.get("cwd", ""),
                mcp_servers=params.get("mcpServers", []),
                config_values={k: str(v["default"]) for k, v in self._config_specs.items()},
                available_commands=session.build_default_commands(),
                runtime_capabilities=self._runtime_capabilities,
                session_id=session_id,
            )
            await self._setup_mcp_if_needed(session_state, params)
            await self._storage.save_session(session_state)

    return ProtocolOutcome(response=response_msg)


async def _handle_session_load(self, message: ACPMessage) -> ProtocolOutcome:
    params = message.params or {}
    session_id = params.get("sessionId")
    if isinstance(session_id, str):
        session_obj = await self._storage.load_session(session_id)
        if session_obj is not None:
            session_obj.runtime_capabilities = self._runtime_capabilities
            await self._setup_mcp_if_needed(session_obj, params)

    return session.session_load(
        message.id, params, self._require_auth, self._authenticated,
        self._config_specs, self._auth_methods, self._storage,
    )


async def _setup_mcp_if_needed(
    self,
    session_state: SessionState,
    params: dict[str, Any],
) -> None:
    """Единая точка инициализации MCP серверов — не дублируется."""
    mcp_servers = params.get("mcpServers", [])
    if mcp_servers and isinstance(mcp_servers, list):
        await self._initialize_mcp_servers(session_state, mcp_servers)
```

### Шаг 5 — Добавить поддержку middleware (опционально)

Если нужна сквозная логика (логирование, метрики, auth-check) — ввести middleware:

```python
from typing import Protocol

class MiddlewareFn(Protocol):
    async def __call__(
        self,
        message: ACPMessage,
        next_handler: MethodHandler,
    ) -> ProtocolOutcome: ...


class ACPProtocol:
    def __init__(self, ..., middleware: list[MiddlewareFn] | None = None) -> None:
        ...
        self._middleware = middleware or []

    async def handle(self, message: ACPMessage) -> ProtocolOutcome:
        if message.method is None:
            return self.handle_client_response(message)

        handler = self._handlers.get(message.method)
        if handler is None:
            ...

        # Применить middleware в обратном порядке (onion pattern)
        wrapped = handler
        for mw in reversed(self._middleware):
            _next = wrapped
            async def wrapped(msg, _h=_next, _mw=mw):
                return await _mw(msg, _h)

        return await wrapped(message)
```

Пример middleware для логирования:

```python
async def logging_middleware(
    message: ACPMessage,
    next_handler: MethodHandler,
) -> ProtocolOutcome:
    logger.debug("handling_method", method=message.method, id=message.id)
    start = time.monotonic()
    result = await next_handler(message)
    elapsed = time.monotonic() - start
    logger.info("method_handled", method=message.method, elapsed_ms=round(elapsed * 1000, 1))
    return result
```

## Тесты

```python
@pytest.mark.asyncio
async def test_unknown_method_returns_method_not_found():
    protocol = ACPProtocol()
    outcome = await protocol.handle(ACPMessage.request("unknown/method", {}))
    assert outcome.response.error is not None
    assert outcome.response.error.code == -32601


@pytest.mark.asyncio
async def test_all_standard_methods_are_registered():
    """Убедиться что все ACP методы зарегистрированы."""
    protocol = ACPProtocol()
    required_methods = [
        "initialize", "authenticate", "session/new", "session/load",
        "session/list", "session/prompt", "session/cancel",
        "session/request_permission_response", "session/set_config_option",
        "ping",
    ]
    for method in required_methods:
        assert method in protocol._handlers, f"Method not registered: {method}"


@pytest.mark.asyncio
async def test_notification_returns_empty_outcome():
    protocol = ACPProtocol()
    msg = ACPMessage(jsonrpc="2.0", method="unknown/notify")
    msg.is_notification = True
    outcome = await protocol.handle(msg)
    assert outcome.response is None
```
