# 2.3 — Замена ручной сериализации в `JsonFileStorage` на Pydantic

**Приоритет:** 🟠 Высокий  
**Оценка:** 1.5 дня  
**Файл:** `src/codelab/server/storage/json_file.py`

---

## Проблема

`JsonFileStorage` содержит ~250 строк ручной сериализации/десериализации через 8 методов `_serialize_*` / `_deserialize_*`. Проблемы:

- При добавлении нового поля в `SessionState` нужно синхронно обновлять оба метода.
- Поле `pending_permission_requests: dict[JsonRpcId, asyncio.Future]` не исключается явно при сериализации — потенциальный `TypeError`.
- Ручной код содержит silent fallback `except Exception: append(entry)` при десериализации, что маскирует повреждение данных.
- Дублирование с Pydantic-логикой, уже встроенной в используемые модели (`HistoryMessage`, `AvailableCommand`, `PlanStep`).

---

## Решение

Перевести `SessionState` с `dataclass` на `pydantic.BaseModel` и использовать встроенную сериализацию.

### Шаг 1 — Перевести `SessionState` на Pydantic

```python
# БЫЛО (state.py):
from dataclasses import dataclass, field

@dataclass(slots=True)
class SessionState:
    session_id: str
    cwd: str
    mcp_servers: list[dict[str, Any]]
    title: str | None = None
    updated_at: str = field(default_factory=lambda: datetime.now(UTC).isoformat())
    history: list[HistoryMessage | dict[str, Any]] = field(default_factory=list)
    pending_permission_requests: dict[JsonRpcId, asyncio.Future] = field(default_factory=dict)
    mcp_manager: Any = None
    ...
```

```python
# СТАЛО (state.py):
from __future__ import annotations

from datetime import UTC, datetime
from typing import Any, ClassVar

from pydantic import BaseModel, Field, field_serializer
import asyncio

from ..messages import JsonRpcId
from ..models import AvailableCommand, HistoryMessage, PlanStep


class SessionState(BaseModel):
    """Персистируемое состояние ACP-сессии."""

    model_config = {"arbitrary_types_allowed": True}

    session_id: str
    cwd: str
    mcp_servers: list[dict[str, Any]] = Field(default_factory=list)
    title: str | None = None
    updated_at: str = Field(default_factory=lambda: datetime.now(UTC).isoformat())
    config_values: dict[str, str] = Field(default_factory=dict)
    history: list[HistoryMessage | dict[str, Any]] = Field(default_factory=list)
    events_history: list[dict[str, Any]] = Field(default_factory=list)
    active_turn: ActiveTurnState | None = None
    tool_call_counter: int = 0
    tool_calls: dict[str, ToolCallState] = Field(default_factory=dict)
    available_commands: list[AvailableCommand | dict[str, Any]] = Field(default_factory=list)
    latest_plan: list[PlanStep | dict[str, Any]] = Field(default_factory=list)
    permission_policy: dict[str, str] = Field(default_factory=dict)
    cancelled_permission_requests: set[JsonRpcId] = Field(default_factory=set)
    cancelled_client_rpc_requests: set[JsonRpcId] = Field(default_factory=set)
    runtime_capabilities: ClientRuntimeCapabilities | None = None

    # --- Runtime-поля: не сериализуются ---

    # asyncio.Future не персистируется — хранится в PendingRequestRegistry (задача 1.5)
    # mcp_manager не персистируется — пересоздаётся при session/load
    mcp_manager: Any = Field(default=None, exclude=True)

    @field_serializer("cancelled_permission_requests", "cancelled_client_rpc_requests")
    def serialize_set(self, value: set) -> list:
        """set не сериализуется в JSON напрямую — конвертируем в list."""
        return list(value)

    @classmethod
    def from_json_data(cls, data: dict[str, Any]) -> SessionState:
        """Десериализовать из JSON, восстанавливая set из list."""
        # Pydantic v2 автоматически конвертирует list → set при валидации
        return cls.model_validate(data)
```

### Шаг 2 — Переписать `JsonFileStorage` используя Pydantic

```python
# БЫЛО (json_file.py): 8 методов _serialize_* / _deserialize_* (~250 строк)

# СТАЛО (json_file.py): ~80 строк
class JsonFileStorage(SessionStorage):
    def __init__(self, base_path: Path | str) -> None:
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)

    def _session_file_path(self, session_id: str) -> Path:
        safe_id = session_id.replace("/", "_").replace("\\", "_").replace("..", "_")
        return self.base_path / f"{safe_id}.json"

    async def save_session(self, session: SessionState) -> None:
        try:
            session.updated_at = datetime.now(UTC).isoformat()
            file_path = self._session_file_path(session.session_id)

            # model_dump(mode="json") — корректно конвертирует все типы включая set → list
            data = session.model_dump(mode="json")

            async with aiofiles.open(file_path, "w", encoding="utf-8") as f:
                await f.write(json.dumps(data, indent=2, ensure_ascii=False))

        except Exception as e:
            raise StorageError(f"Failed to save session {session.session_id}: {e}") from e

    async def load_session(self, session_id: str) -> SessionState | None:
        try:
            file_path = self._session_file_path(session_id)
            if not file_path.exists():
                return None

            async with aiofiles.open(file_path, encoding="utf-8") as f:
                content = await f.read()

            data = json.loads(content)
            return SessionState.model_validate(data)

        except json.JSONDecodeError as e:
            raise StorageError(f"Corrupted session file {session_id}") from e
        except ValidationError as e:
            raise StorageError(f"Invalid session data {session_id}: {e}") from e
        except Exception as e:
            raise StorageError(f"Failed to load session {session_id}: {e}") from e

    # ... остальные методы аналогично
```

### Шаг 3 — Обработать backward compatibility

Существующие JSON-файлы могут не иметь новых полей, добавленных после первой записи. Pydantic обработает это через `default_factory`, но нужно убедиться в корректности схемы:

```python
# Добавить схему версии для будущих миграций:
class SessionState(BaseModel):
    schema_version: int = Field(default=1, exclude=False)
    ...
```

Для автоматической миграции старых файлов — добавить `model_validator`:

```python
from pydantic import model_validator

@model_validator(mode="before")
@classmethod
def migrate_schema(cls, data: dict[str, Any]) -> dict[str, Any]:
    version = data.get("schema_version", 0)
    if version < 1:
        # Мигрировать поля из v0 в v1
        data.setdefault("events_history", [])
        data["schema_version"] = 1
    return data
```
