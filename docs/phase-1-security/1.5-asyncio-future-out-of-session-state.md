# 1.5 — Вынести `asyncio.Future` из `SessionState`

**Приоритет:** 🟠 Высокий  
**Оценка:** 5 часов  
**Файлы:** `src/codelab/server/protocol/state.py`, `src/codelab/server/storage/json_file.py`, `src/codelab/server/protocol/core.py`

---

## Проблема

`SessionState` содержит поле с `asyncio.Future`:

```python
# state.py
@dataclass(slots=True)
class SessionState:
    ...
    pending_permission_requests: dict[JsonRpcId, asyncio.Future] = field(default_factory=dict)
```

`asyncio.Future` — runtime-объект, привязанный к event loop. Он **не сериализуется в JSON**. `JsonFileStorage._serialize_session()` при попытке сохранить сессию с незавершёнными futures либо упадёт с `TypeError`, либо (если есть try/except) молча потеряет данные.

Кроме того, `asyncio.Future` не имеет смысла в персистируемом состоянии — при перезагрузке сервера event loop пересоздаётся, старые futures становятся недействительными.

---

## Решение

Вынести `asyncio.Future` из `SessionState` в отдельный runtime-реестр на уровне `ACPProtocol`.

### Шаг 1 — Удалить поле из `SessionState`

```python
# БЫЛО (state.py):
@dataclass(slots=True)
class SessionState:
    ...
    pending_permission_requests: dict[JsonRpcId, asyncio.Future] = field(default_factory=dict)
    mcp_manager: Any = None
```

```python
# СТАЛО (state.py):
@dataclass(slots=True)
class SessionState:
    ...
    # pending_permission_requests удалено — хранится в PendingRequestRegistry
    mcp_manager: Any = None
```

### Шаг 2 — Создать `PendingRequestRegistry`

Создать файл `src/codelab/server/protocol/pending_registry.py`:

```python
"""Реестр ожидающих asyncio.Future для permission requests.

Хранит runtime-объекты, которые не могут быть сериализованы
и не должны присутствовать в персистируемом SessionState.
"""

from __future__ import annotations

import asyncio
from typing import Any

import structlog

from ..messages import JsonRpcId

logger = structlog.get_logger()


class PendingRequestRegistry:
    """Хранилище asyncio.Future для ожидающих permission requests.

    Жизненный цикл: создаётся в ACPProtocol, не персистируется.
    При перезапуске сервера — пересоздаётся пустым.
    """

    def __init__(self) -> None:
        self._futures: dict[JsonRpcId, asyncio.Future[Any]] = {}

    def create(self, request_id: JsonRpcId) -> asyncio.Future[Any]:
        """Создать и зарегистрировать Future для request_id."""
        loop = asyncio.get_running_loop()
        future: asyncio.Future[Any] = loop.create_future()
        self._futures[request_id] = future
        logger.debug("pending_request_created", request_id=request_id)
        return future

    def resolve(self, request_id: JsonRpcId, result: Any) -> bool:
        """Завершить Future с результатом. Возвращает True если Future найден."""
        future = self._futures.pop(request_id, None)
        if future is None:
            return False
        if not future.done():
            future.set_result(result)
            logger.debug("pending_request_resolved", request_id=request_id)
        return True

    def cancel(self, request_id: JsonRpcId) -> bool:
        """Отменить Future. Возвращает True если Future найден."""
        future = self._futures.pop(request_id, None)
        if future is None:
            return False
        if not future.done():
            future.cancel()
            logger.debug("pending_request_cancelled", request_id=request_id)
        return True

    def has(self, request_id: JsonRpcId) -> bool:
        """Проверить наличие ожидающего Future."""
        return request_id in self._futures

    def cancel_all(self) -> int:
        """Отменить все ожидающие futures. Возвращает количество отменённых."""
        count = 0
        for request_id in list(self._futures.keys()):
            if self.cancel(request_id):
                count += 1
        return count

    def __len__(self) -> int:
        return len(self._futures)
```

### Шаг 3 — Обновить `ACPProtocol` для использования реестра

```python
# core.py

from .pending_registry import PendingRequestRegistry

class ACPProtocol:
    def __init__(self, ...) -> None:
        ...
        # Runtime-реестр futures — не персистируется, не входит в SessionState
        self._pending_registry = PendingRequestRegistry()
```

### Шаг 4 — Обновить места использования `pending_permission_requests`

Найти все обращения к `session.pending_permission_requests` через:

```bash
grep -rn "pending_permission_requests" src/codelab/
```

Заменить:

```python
# БЫЛО:
future = asyncio.get_event_loop().create_future()
session.pending_permission_requests[request_id] = future
result = await future

# СТАЛО:
future = self._pending_registry.create(request_id)
result = await future
```

```python
# БЫЛО (при получении ответа):
future = session.pending_permission_requests.pop(request_id, None)
if future and not future.done():
    future.set_result(permission_result)

# СТАЛО:
self._pending_registry.resolve(request_id, permission_result)
```

### Шаг 6 — Обработка перезапуска сервера

При перезапуске `PendingRequestRegistry` пересоздаётся пустым. Сессии с незавершёнными permission requests (где пользователь не успел ответить до рестарта) будут иметь `active_turn.permission_request_id`, но без соответствующего Future.

Добавить восстановление при `session/load`:

```python
# В handle() при session/load — если есть active_turn с permission_request_id,
# но Future не зарегистрирован — автоматически отменить turn:
if session.active_turn and session.active_turn.permission_request_id:
    if not self._pending_registry.has(session.active_turn.permission_request_id):
        logger.warning(
            "session_loaded_with_orphaned_permission_request",
            session_id=session_id,
            permission_request_id=session.active_turn.permission_request_id,
        )
        # Сбросить активный turn — он не может быть продолжен
        session.active_turn = None
```

## Тесты

```python
import asyncio
import pytest
from codelab.server.protocol.pending_registry import PendingRequestRegistry


@pytest.mark.asyncio
async def test_create_and_resolve():
    registry = PendingRequestRegistry()
    future = registry.create("req-1")
    assert registry.has("req-1")

    registry.resolve("req-1", {"outcome": "allow"})
    result = await future
    assert result == {"outcome": "allow"}
    assert not registry.has("req-1")


@pytest.mark.asyncio
async def test_cancel():
    registry = PendingRequestRegistry()
    future = registry.create("req-1")

    registry.cancel("req-1")
    assert future.cancelled()
    assert not registry.has("req-1")


@pytest.mark.asyncio
async def test_cancel_all():
    registry = PendingRequestRegistry()
    f1 = registry.create("req-1")
    f2 = registry.create("req-2")

    count = registry.cancel_all()
    assert count == 2
    assert f1.cancelled()
    assert f2.cancelled()


def test_resolve_unknown_returns_false():
    registry = PendingRequestRegistry()
    result = registry.resolve("unknown-id", {})
    assert result is False
```
