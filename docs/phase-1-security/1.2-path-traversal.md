# 1.2 — Path Traversal в `FileSystemExecutor._validate_path()`

**Приоритет:** 🟠 Высокий  
**Оценка:** 1 час  
**Файл:** `src/codelab/client/infrastructure/services/file_system_executor.py`

---

## Проблема

Метод `_validate_path()` проверяет, находится ли путь внутри `base_path`, используя строковый метод `startswith`:

```python
if not file_path_str.startswith(base_str):
    raise ValueError(f"Path traversal detected: {path} outside {base_resolved}")
```

Эта проверка **ненадёжна**. Рассмотрим сценарий:

```
base_path  = /home/user/projects
evil_path  = /home/user/projects_evil/secret.txt
```

Строка `/home/user/projects_evil/secret.txt` начинается с `/home/user/projects` — проверка пройдёт, хотя путь находится **вне** sandbox. Атакующий может:

1. Прочитать любой файл вне рабочей директории (`~/.ssh/id_rsa`, `~/.aws/credentials`).
2. Перезаписать произвольные файлы.

Вектор: агент получает задачу "прочитай конфиг" и формирует путь `../../../etc/passwd` или `../projects_evil/secret`.

---

## Решение

Заменить `startswith` на `Path.is_relative_to()` (Python ≥ 3.9). Этот метод работает с компонентами пути, а не со строками — он не подвержен описанной атаке.

**Файл:** `src/codelab/client/infrastructure/services/file_system_executor.py`

```python
# БЫЛО:
def _validate_path(self, path: str) -> Path:
    try:
        if self.base_path:
            base_resolved = self.base_path.resolve()
            file_path = (base_resolved / path).resolve()
        else:
            file_path = Path(path).resolve()
    except (ValueError, RuntimeError) as e:
        raise ValueError(f"Invalid path: {path}") from e

    if self.base_path:
        base_resolved = self.base_path.resolve()
        file_path_str = str(file_path)
        base_str = str(base_resolved)

        # УЯЗВИМОСТЬ: startswith не учитывает границы компонентов пути
        if not file_path_str.startswith(base_str):
            raise ValueError(f"Path traversal detected: {path} outside {base_resolved}")

    return file_path
```

```python
# СТАЛО:
def _validate_path(self, path: str) -> Path:
    """Валидировать и нормализовать путь. Защита от path traversal.

    Args:
        path: Путь к файлу (абсолютный или относительный от base_path)

    Returns:
        Нормализованный абсолютный Path

    Raises:
        ValueError: Некорректный путь или попытка выйти за пределы base_path
    """
    try:
        if self.base_path:
            base_resolved = self.base_path.resolve()
            file_path = (base_resolved / path).resolve()
        else:
            file_path = Path(path).resolve()
    except (ValueError, RuntimeError) as e:
        logger.warning("invalid_path", path=path, error=str(e))
        raise ValueError(f"Invalid path: {path}") from e

    if self.base_path:
        base_resolved = self.base_path.resolve()

        # is_relative_to() проверяет компоненты пути, а не строковый префикс.
        # /home/user/projects_evil НЕ является относительным к /home/user/projects
        if not file_path.is_relative_to(base_resolved):
            msg = f"Path traversal detected: '{path}' resolves outside sandbox '{base_resolved}'"
            logger.warning(
                "path_traversal_attempt",
                path=path,
                resolved=str(file_path),
                base_path=str(base_resolved),
            )
            raise ValueError(msg)

    return file_path
```

### Совместимость с Python 3.8

Если проект должен работать на Python 3.8 (где `is_relative_to` недоступен), используйте альтернативу:

```python
# Совместимый вариант для Python 3.8:
try:
    file_path.relative_to(base_resolved)
except ValueError:
    raise ValueError(f"Path traversal detected: '{path}' outside '{base_resolved}'")
```

`relative_to()` доступен с Python 3.0 и выбрасывает `ValueError`, если путь не является дочерним.

## Дополнительная защита: проверка симлинков

`resolve()` раскрывает симлинки, что защищает от атак вида:

```bash
ln -s /etc/passwd /home/user/projects/evil_link
# executor.read_text_file("evil_link")
```

Однако если файл создаётся агентом (`write_text_file`), а потом читается — симлинк уже будет внутри `base_path`. Для полной защиты добавить проверку после `resolve()`:

```python
# После resolve() — убедиться что реальный путь тоже внутри sandbox:
if self.base_path and not file_path.is_relative_to(base_resolved):
    raise ValueError(f"Symlink escape detected: '{path}'")
```

## Тесты

Добавить в `tests/client/test_file_system_executor.py`:

```python
import pytest
from pathlib import Path
from codelab.client.infrastructure.services.file_system_executor import FileSystemExecutor


@pytest.fixture
def sandbox(tmp_path: Path) -> Path:
    """Создаёт sandbox директорию с тестовым файлом."""
    (tmp_path / "safe.txt").write_text("safe content")
    return tmp_path


def test_read_file_inside_sandbox(sandbox: Path):
    executor = FileSystemExecutor(base_path=sandbox)
    # Должно работать без ошибок
    import asyncio
    content = asyncio.run(executor.read_text_file("safe.txt"))
    assert content == "safe content"


def test_path_traversal_dotdot_blocked(sandbox: Path):
    """../../../etc/passwd должен быть заблокирован."""
    executor = FileSystemExecutor(base_path=sandbox)
    with pytest.raises(ValueError, match="Path traversal"):
        executor._validate_path("../../../etc/passwd")


def test_path_traversal_similar_name_blocked(sandbox: Path, tmp_path: Path):
    """Директория с похожим именем должна быть заблокирована."""
    # Создаём директорию рядом с sandbox с похожим именем
    evil_dir = tmp_path.parent / (tmp_path.name + "_evil")
    evil_dir.mkdir(exist_ok=True)
    (evil_dir / "secret.txt").write_text("secret")

    executor = FileSystemExecutor(base_path=sandbox)
    with pytest.raises(ValueError, match="Path traversal"):
        # Попытка выйти в директорию с похожим именем
        executor._validate_path(str(evil_dir / "secret.txt"))


def test_absolute_path_inside_sandbox_allowed(sandbox: Path):
    """Абсолютный путь внутри sandbox должен быть разрешён."""
    executor = FileSystemExecutor(base_path=sandbox)
    result = executor._validate_path(str(sandbox / "safe.txt"))
    assert result == (sandbox / "safe.txt").resolve()


def test_no_sandbox_allows_absolute_paths():
    """Без sandbox абсолютные пути разрешены."""
    executor = FileSystemExecutor(base_path=None)
    result = executor._validate_path("/tmp")
    assert result == Path("/tmp").resolve()
```
